`
  This script will match numbers that are between 0 and 9.
`

("x<-" "source" ("x->" "@"))
("x<-" "destination" "")

("?"
  ("&" (">" ("x->" "source") #-1#) ("<" ("x->" "source") #10#))
  `Find the ones word`
  ("x<-" "destination"
    ("$->" {
      "0" "zero"
      "1" "one"
      "2" "two"
      "3" "three"
      "4" "four"
      "5" "five"
      "6" "six"
      "7" "seven"
      "8" "eight"
      "9" "nine"
    } ("\"" ("x->" "source")))
  )
  ?
)

("x->" "destination")
