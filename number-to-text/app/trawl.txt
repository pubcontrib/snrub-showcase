`
  This script will run a sub-script and set any catches to a global variable.
`

("x<-" "net" ("$->" ("x->" "@") #1#))
("x<-" "bait" ("$->" ("x->" "@") #2#))
("x<-" "icebox" ("$->" ("x->" "@") #3#))

("x<-" "catch" ("~" ("x->" ("x->" "net")) ("x->" "bait")))

("?" ("=" ("x->" "catch") "") ? ("x<-" ("x->" "icebox") ("x->" "catch")))
