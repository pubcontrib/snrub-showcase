`
  This script will check if two objects are equal otherwise throw an error.
`

("x<-" "expected" ("$->" ("x->" "@") #1#))
("x<-" "actual" ("$->" ("x->" "@") #2#))

("?"
  ("!" ("=" ("x->" "expected") ("x->" "actual")))
  ("..."
    ("x<-" "message" ("+" "[ERROR] Assert failed. Expected: "
        (";" ("x->" "expected"))))
    ("x<-" "message" ("+" ("x->" "message")
        ". Actual: "))
    ("x<-" "message" ("+" ("x->" "message")
        (";" ("x->" "actual"))))
    ("x<-" "message" ("+" ("x->" "message")
        "."))
    ("<>" ("x->" "message"))
  )
  ?
)
